# 1. Создать .env файл
cd ~/autobid-tj

NODE_ENV=production
PORT=3001
DATABASE_URL=postgresql://autobid_user:AutoBid2025@localhost:5432/autobid_db
EOF

# 2. Проверить все службы

systemctl status postgresql --no-pager -l | head -3

# 3. Проверить подключение к базе данных
sudo -u postgres psql -d autobid_db -c "SELECT version();" | head -1
